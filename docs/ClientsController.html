def index
  if melfrag=params[:melfrag]
    allclients = Client.where("mel LIKE ?", "%#{melfrag}%")
    results=[]
    allclients.each do |c|
      results<< {"email": c.mel,"id": c.id}
    end
  else
    results=Client.all
  end
  render json: results
end
